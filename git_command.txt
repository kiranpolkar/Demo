#Specify name
git config --global user.name "Kiran Polkar"

#Specify email
git config --global user.email kiranpolkar2812@gmail.com

#Specify default branch
git config --global init.default branch main

#Go to code directory 
cd G:\GitTutorial

#Initialize this directory in git
git init

#To get repository status - list the files which are committed or yet to be commited
git status

#Adding file to Git to track the file
git add Sample.txt

#Untrack the file
git rm --cached Sample.txt

#To ignore the files from git showup - 1)create file with .gitignore extention 2) Open it in note pad and specify file name or extentions to ignore

#Track all the files 
git add --all # or git add -A or git add .

#Commiting files in git
git commit -m "First commit - commting all files in git repository"

#To see modification in a file 
git diff Sample.txt

#To add modified file in staging
git add Sample.txt

#To remove file from staging 
git restore --staged Sample.txt

#To Skip staging and direct commit the modified file 
git commit -a -m "Updated file to add new code"
